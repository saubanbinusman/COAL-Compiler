METHOD sum TAKES x AS INTEGER, RETURNS INTEGER
	LET retVal AS INTEGER
	LET argToFunc AS INTEGER
	
	IF (x == 0)
		RETURN 0
	ELSE
		SET argToFunc = x - 1
		SET retVal = CALL sum(argToFunc)
		SET retVal = retVal + x
		RETURN retVal
	ENDIF
ENDMETHOD

METHOD MAIN RETURNS NOTHING
	LET finalValue AS INTEGER
	SET finalValue = CALL sum(6)

	PRINTLINE finalValue
ENDMETHOD
